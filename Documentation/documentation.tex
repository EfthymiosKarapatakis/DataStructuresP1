	\subsection{Λογική}
	Η αναζήτηση γίνεται συγκρίνοντας το κλειδί με τη ρίζα και κινούμαστε αριστερά ή δεξιά.
	
	\subsection{Κώδικας}
	\begin{lstlisting}
		// Παράδειγμα BST search
		struct Node* binaryTreeSearch(struct Node* root, int key) {
			if(root==NULL) return NULL;
			if(root->data==key) return root;
			if(key<root->data) return binaryTreeSearch(root->left,key);
			else return binaryTreeSearch(root->right,key);
		}
	\end{lstlisting}

	\section{Άσκηση 3: Διαπεράσεις Δέντρου}
	
	\subsection{Λογική}
	\begin{itemize}
		\item Preorder: Ρίζα-Αριστερά-Δεξιά
		\item Inorder: Αριστερά-Ρίζα-Δεξιά
		\item Postorder: Αριστερά-Δεξιά-Ρίζα
		\item Level Order: Επίπεδο προς επίπεδο από αριστερά προς δεξιά
	\end{itemize}
	
	\subsection{Κώδικας}
	\begin{lstlisting}
		// Preorder example
		void preorder(struct Node* root){
			if(root==NULL) return;
			printf("%d ",root->data);
			preorder(root->left);
			preorder(root->right);
		}
	\end{lstlisting}

		struct Node* buildTreeFromPre(char inorder[], char preorder[], int left, int right, int* preIndex){
			if(left>right) return NULL;
			char rootVal = preorder[*preIndex];
			(*preIndex)++;
			struct Node* root = createNode(rootVal);
			int index = search(inorder, rootVal, left, right);
			root->left = buildTreeFromPre(inorder, preorder, left, index-1, preIndex);
			root->right = buildTreeFromPre(inorder, preorder, index+1, right, preIndex);
			return root;
		}
